genes<-subset(gff[[47]],type=="gene")
plot(genes$start)

head(genes)
distance<-vector()
for(i in 1:(length(genes$start)-1)){
  distance[i]<-(genes$end[i]-genes$start[i+1])
}
starts<-genes$start
names(starts)<-gsub(pattern = ".*ID=(.*);.*" , replacement = "\\1",genes$attributes,perl=T)
# Ward Hierarchical Clustering
d <- dist(starts, method = "euclidean") # distance matrix
fit <- hclust(d, method="ward")
The "ward" method has been renamed to "ward.D"; note new "ward.D2"
plot(fit,cex=.6) # display dendogram
groups <- cutree(fit, k=10) # cut tree into 5 clusters
# draw dendogram with red borders around the 5 clusters
rect.hclust(fit, k=10, border="red")
